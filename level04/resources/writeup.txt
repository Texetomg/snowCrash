use CGI qw{param}; # (Общий интерфейс шлюза) - стандарт интерфейса, используемого для связи внешней программы с веб-сервером, в нашем случае мы еще получаем параметр
print "Content-type: text/html\n\n"; # Задаем HTTP заголовки

  $y = $_[0]; # Присваиваем пременной y значение из массива
  print `echo $y 2>&1`; # Выводим значение переменной y
}
x(param("x")); # Вызываем функцию x с заданным параметром от ключа "x"
